include(ExternalProject)
find_package(Git REQUIRED)

ExternalProject_Add(
  Hypodermic

  GIT_REPOSITORY "https://github.com/ybainier/Hypodermic.git"
  GIT_TAG "v2.4"

  INSTALL_COMMAND cmake -E echo "Skipping install step."
  
  SOURCE_DIR "${CMAKE_SOURCE_DIR}/externals/hypodermic"
)
set(Hypodermic_INCLUDE_DIRS "${CMAKE_SOURCE_DIR}/externals/hypodermic")
include_directories(${Hypodermic_INCLUDE_DIRS})


ExternalProject_Add(
    Doctest
    PREFIX ${CMAKE_SOURCE_DIR}/externals/doctest
    GIT_REPOSITORY https://github.com/onqtam/doctest.git
    TIMEOUT 10
    UPDATE_COMMAND ${GIT_EXECUTABLE} pull
    CONFIGURE_COMMAND ""
    BUILD_COMMAND ""
    INSTALL_COMMAND ""
    LOG_DOWNLOAD ON
)

# Expose required variable (DOCTEST_INCLUDE_DIR) to parent scope
ExternalProject_Get_Property(Doctest source_dir)
set(Doctest_INCLUDE_DIR ${source_dir}/doctest CACHE INTERNAL "Path to include folder for doctest")
include_directories(${Doctest_INCLUDE_DIR})
